    // src/components/forms/FormField.astro
    // Phase 2 form primitive — consumed by RetreatApplicationForm (Plan 04) + ContactForm (Plan 05).
    // Styles per UI-SPEC §"Form field states" — padding 12px 16px, radius 8px, font-size 17px.
    // Touch target ~48px exceeds WCAG 2.5.5 44×44 requirement.

    interface Props {
      type: 'text' | 'email' | 'tel' | 'textarea' | 'select' | 'radio';
      name: string;
      label: string;
      required?: boolean;
      helpText?: string;
      rows?: number;
      options?: string[];
    }
    const { type, name, label, required = false, helpText, rows = 3, options = [] } = Astro.props as Props;
    ---
    <div class="form-field">
      {type === 'radio' ? (
        <fieldset>
          <legend>
            {label}{required && <span class="required" aria-hidden="true"> *</span>}
          </legend>
          <div class="radio-group">
            {options.map((opt, i) => (
              <label class="radio-option">
                <input type="radio" name={name} value={opt} required={required && i === 0 ? true : undefined} />
                <span>{opt}</span>
              </label>
            ))}
          </div>
        </fieldset>
      ) : (
        <>
          <label for={name}>
            {label}{required && <span class="required" aria-hidden="true"> *</span>}
          </label>
          {type === 'textarea' ? (
            <textarea id={name} name={name} rows={rows} required={required}></textarea>
          ) : type === 'select' ? (
            <select id={name} name={name} required={required}>
              {options.map(opt => <option value={opt}>{opt}</option>)}
            </select>
          ) : (
            <input id={name} name={name} type={type} required={required} autocomplete={type === 'email' ? 'email' : type === 'tel' ? 'tel' : undefined} />
          )}
        </>
      )}
      {helpText && <p class="help-text">{helpText}</p>}
    </div>
