```css
/* src/styles/tokens.css
 * Phase 1 design tokens. Implements:
 *   D-01: Dark only — no light/dark toggle
 *   D-02: Saturated psychedelic palette (locked hex)
 *   D-03: Fraunces (display, 300/500) + Inter (body, 400/500)

 *
 * Tailwind v4 @theme block: tokens become utilities automatically (e.g. bg-orange-mystic, font-display).
 */

@theme {
  /* === Brand palette (locked: D-02) === */
  --color-orange-mystic: #ff7a3d;
  --color-violet-deep:   #5b2a86;
  --color-blue-cosmic:   #0b1d4f;

  /* === Surfaces === */
  --color-bg-base:       #050a1a;            /* dark navy near-black */
  --color-bg-elevated:   #0b1428;            /* one step above bg for cards/header on scroll */

  /* === Typography colors === */
  --color-text-primary:  #f4ede3;            /* warm cream — NOT pure white */
  --color-text-muted:    rgba(244, 237, 227, 0.65);
  --color-text-subtle:   rgba(244, 237, 227, 0.45);

  /* === Chrome / borders === */
  --color-chrome-border: rgba(244, 237, 227, 0.08);
  --color-chrome-border-strong: rgba(244, 237, 227, 0.16);

  /* === Font families (D-03) === */
  --font-display: 'Fraunces', 'Times New Roman', serif;
  --font-body:    'Inter', system-ui, -apple-system, 'Segoe UI', sans-serif;

  /* === Spacing rhythm === */
  --space-section: clamp(4rem, 10vw, 8rem);
}

/* === Animated gradient angle: registered as a typed custom property so it interpolates (D-04) === */
@property --cosmic-angle {
  syntax: '<angle>';
  initial-value: 0deg;
  inherits: false;
}

/* === Reusable cosmic-rotate keyframe (D-04) === */
@keyframes cosmic-rotate {
  to {
    --cosmic-angle: 360deg;
  }
}

/* === Hero background utility — apply to a fixed-positioned div behind hero content === */
.hero-bg {
  background: conic-gradient(
    from var(--cosmic-angle),
    var(--color-violet-deep),
    var(--color-orange-mystic),
    var(--color-blue-cosmic),
    var(--color-violet-deep)
  );
  filter: blur(80px) saturate(1.2);
  animation: cosmic-rotate 32s linear infinite;
}

/* === D-04 mandatory: reduced-motion users get no animation === */
@media (prefers-reduced-motion: reduce) {
  .hero-bg {
    animation: none;
  }
}

/* === D-05 optional: SVG noise overlay class — used by Hero in Plan 04 === */
.noise-overlay {
  position: absolute;
  inset: 0;
  pointer-events: none;
  opacity: 0.08;
  mix-blend-mode: overlay;

}
```

```css
/* src/styles/global.css
 * Single global stylesheet imported by BaseLayout. Order matters:
 *   1. Tailwind base/components/utilities (provides reset + @theme registration)
 *   2. Tokens (declares @theme — must be available before utilities resolve)
 *   3. Google Fonts (display=swap to avoid CLS — PITFALLS#8)
 *   4. Base body styles (dark only — D-01)
 */

@import "tailwindcss";
@import "./tokens.css";

/* D-03: Fraunces (display) + Inter (body), pesos mínimos para presupuesto perf */
@import url("https://fonts.googleapis.com/css2?family=Fraunces:opsz,wght@9..144,300;9..144,500&family=Inter:wght@400;500&display=swap");

/* Dark-only baseline (D-01) */
html {
  background-color: var(--color-bg-base);
  color: var(--color-text-primary);
  -webkit-text-size-adjust: 100%;
  scroll-behavior: smooth;
}

body {
  margin: 0;
  font-family: var(--font-body);
  font-weight: 400;
  line-height: 1.6;
  background-color: var(--color-bg-base);
  color: var(--color-text-primary);
  min-height: 100vh;
  -webkit-font-smoothing: antialiased;
  -moz-osx-font-smoothing: grayscale;
}

h1, h2, h3, h4, h5, h6 {
  font-family: var(--font-display);
  font-weight: 300;
  line-height: 1.2;
  letter-spacing: -0.01em;
  margin: 0;
}

a {
  color: inherit;
  text-decoration: none;
}

a:hover, a:focus-visible {
  color: var(--color-orange-mystic);
  transition: color 200ms ease;
}

/* Honor users with prefers-reduced-motion: reduce — disable smooth scroll + transitions on chrome */
@media (prefers-reduced-motion: reduce) {
  html {
    scroll-behavior: auto;
  }
  *, *::before, *::after {
    animation-duration: 0.001ms !important;
    animation-iteration-count: 1 !important;
    transition-duration: 0.001ms !important;

}

/* Container helper — Plan 04 sections will use this */
.container {
  max-width: 1200px;
  margin-inline: auto;
  padding-inline: clamp(1.25rem, 4vw, 2rem);
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Google Fonts CDN → browser | Third-party font asset loaded on every page; CDN compromise would inject styles. Acceptable risk for MVP (industry standard). |
| Tailwind compile pipeline → emitted CSS | Build-time only; no runtime tampering surface. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-06 | Information Disclosure | Google Fonts `?display=swap` request leaks IP of every visitor to Google | accept | Documented in STACK.md as MVP acceptable; Phase 2 can self-host if Lighthouse perf or privacy posture requires it. No PII transmitted. |
| T-01-07 | Tampering | Inline `data:image/svg+xml` noise overlay payload | mitigate | Payload is a static `<feTurbulence>` filter authored in this plan — no user-controlled input flows into it. |
| T-01-08 | Denial of Service | `prefers-reduced-motion: reduce` ignored → vestibular harm to sensitive users | mitigate | Mandatory media query in tokens.css AND global.css (Task 2 + Task 3 acceptance criteria both grep for it). |
| T-01-09 | Tampering | Tailwind v4 + Astro version drift breaks `@theme` resolution | mitigate | `npx astro add tailwind --yes` pins matching versions of `tailwindcss` + `@tailwindcss/vite`; Task 1 acceptance criteria verifies both are at `^4.x`. |
</threat_model>
